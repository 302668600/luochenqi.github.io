  ];

  var dpr = window.devicePixelRatio || 1;
  var W = canvas.parentElement.clientWidth - 36;
  var TOTAL_H = 380;
  var K_H = 240, V_H = 80, MA_H = 40;
  var PAD_L = 50, PAD_R = 10, PAD_TOP = 20, PAD_BOT = 4;

  canvas.width = W * dpr;
  canvas.height = TOTAL_H * dpr;
  canvas.style.width = W + 'px';
  canvas.style.height = TOTAL_H + 'px';

  var ctx = canvas.getContext('2d');
  ctx.scale(dpr, dpr);

  var isDark = document.documentElement.classList.contains('dark');
  var BG = isDark ? '#23262f' : '#ffffff';
  var TEXT_C = isDark ? '#9aa0a6' : '#5f6368';
  var GRID_C = isDark ? '#2d3139' : '#f0f2f5';
  var BORDER_C = isDark ? '#2d3139' : '#dadce0';

  ctx.fillStyle = BG;
  ctx.fillRect(0, 0, W, TOTAL_H);

  var n = kdata.length;
  var barW = Math.max(2, Math.floor((W - PAD_L - PAD_R) / n) - 1);
  var gap = Math.floor((W - PAD_L - PAD_R) / n);

  // 计算均线
  function calcMA(arr, period) {
    var res = [];
    for (var i = 0; i < arr.length; i++) {
      if (i < period - 1) { res.push(null); continue; }
      var sum = 0;
      for (var j = i - period + 1; j <= i; j++) sum += arr[j].c;
      res.push(sum / period);
    }
    return res;
  }
  var ma5 = calcMA(kdata, 5), ma10 = calcMA(kdata, 10), ma20 = calcMA(kdata, 20), ma60 = calcMA(kdata, 60);

  var kMaxP = Math.max.apply(null, kdata.map(function(d){return d.h;}));
  var kMinP = Math.min.apply(null, kdata.map(function(d){return d.l;}));
  var vMax = Math.max.apply(null, kdata.map(function(d){return d.v;}));

  var kTop = PAD_TOP, kBot = PAD_TOP + K_H;
  var vTop = kBot + 10, vBot = vTop + V_H;

  function priceY(p) { return kTop + (kMaxP - p) / (kMaxP - kMinP) * (kBot - kTop); }
  function volY(v) { return vBot - v / vMax * (vBot - vTop); }
  function barX(i) { return PAD_L + i * gap + gap / 2; }

  // 绘制网格
  ctx.strokeStyle = GRID_C;

  for (var gi = 0; gi <= 4; gi++) {

    var up = d.c >= d.o;
    var color = up ? '#ea4335' : '#34a853';
    ctx.strokeStyle = color;
    ctx.fillStyle = color;
    ctx.lineWidth = 1;
    // 影线
    ctx.beginPath();
    ctx.moveTo(x, priceY(d.h));
    ctx.lineTo(x, priceY(d.l));
    ctx.stroke();
    // 实体
    var yO = priceY(d.o), yC = priceY(d.c);
    var rectY = Math.min(yO, yC), rectH = Math.max(Math.abs(yO - yC), 1);
    if (up) { ctx.fillRect(x - barW/2, rectY, barW, rectH); }
    else { ctx.strokeRect(x - barW/2, rectY, barW, rectH); }
  });

  // 绘制均线
  function drawMA(ma, color) {
    ctx.beginPath(); ctx.strokeStyle = color; ctx.lineWidth = 1.5;
    var started = false;
    for (var i = 0; i < ma.length; i++) {
      if (ma[i] === null) continue;
      var x = barX(i), y = priceY(ma[i]);
      if (!started) { ctx.moveTo(x, y); started = true; }
      else { ctx.lineTo(x, y); }
    }
    ctx.stroke();
  }
  drawMA(ma5, '#f57c00');
  drawMA(ma10, '#9c27b0');
  drawMA(ma20, '#1a73e8');
  drawMA(ma60, '#ea4335');

  var legends = [{c:'#f57c00',t:'MA5'},{c:'#9c27b0',t:'MA10'},{c:'#1a73e8',t:'MA20'},{c:'#ea4335',t:'MA60'}];

    ctx.fillStyle = TEXT_C; ctx.font = '10px sans-serif'; ctx.textAlign = 'left';
    ctx.fillText(l.t, lx + 18, ly + 4);
  });

  var curY = priceY(493.2);

  ctx.strokeStyle = '#fbbc04'; ctx.lineWidth = 1;
  ctx.beginPath(); ctx.moveTo(PAD_L, curY); ctx.lineTo(W - PAD_R, curY); ctx.stroke();
  ctx.setLineDash([]);
  ctx.fillStyle = '#fbbc04'; ctx.font = 'bold 10px sans-serif'; ctx.textAlign = 'left';
  ctx.fillText('493.2', W - PAD_R + 1, curY + 3);

  // 成交量
  ctx.fillStyle = TEXT_C; ctx.font = '10px sans-serif'; ctx.textAlign = 'left';
  ctx.fillText('Vol(万)', PAD_L, vTop - 2);
  kdata.forEach(function(d, i) {
    var x = barX(i);
    var up = d.c >= d.o;
    ctx.fillStyle = up ? 'rgba(234,67,53,0.6)' : 'rgba(52,168,83,0.6)';
    var vy = volY(d.v), vh = vBot - vy;
    ctx.fillRect(x - barW/2, vy, barW, vh);
  });

  // X轴日期（每10个显示一个）
  ctx.fillStyle = TEXT_C; ctx.font = '9px sans-serif'; ctx.textAlign = 'center';
  for (var xi = 0; xi < n; xi += 10) {
    ctx.fillText(kdata[xi].d, barX(xi), vBot + 12);
  }
  ctx.fillText(kdata[n-1].d, barX(n-1), vBot + 12);

  // 边框
  ctx.strokeStyle = BORDER_C; ctx.lineWidth = 1;
  ctx.strokeRect(PAD_L, kTop, W - PAD_L - PAD_R, kBot - kTop);
  ctx.strokeRect(PAD_L, vTop, W - PAD_L - PAD_R, vBot - vTop);
})();
