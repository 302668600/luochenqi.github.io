for(var i=0;i<n;i++){allP.push(data[i].h,data[i].l);}
var pMin=Math.min.apply(null,allP)-5;
var pMax=Math.max.apply(null,allP)+5;
var pad={l:40,r:8,t:20,b:36};
var cW=W-pad.l-pad.r,cH=H-pad.t-pad.b;
var bw=Math.max(2,Math.floor(cW/n*0.7));
function px(p){return pad.t+cH*(1-(p-pMin)/(pMax-pMin));}
function bx(i){return pad.l+i*(cW/n)+cW/n*0.15;}
ctx.fillStyle='#f8f9fa';
ctx.fillRect(0,0,W,H);
ctx.strokeStyle='#e0e0e0';ctx.lineWidth=0.5;
for(var g=0;g<5;g++){
var gy=pad.t+g*(cH/4);
ctx.beginPath();ctx.moveTo(pad.l,gy);ctx.lineTo(pad.l+cW,gy);ctx.stroke();
var gp=(pMax-(pMax-pMin)*g/4).toFixed(0);
ctx.fillStyle='#999';ctx.font='10px sans-serif';ctx.textAlign='right';
ctx.fillText(gp,pad.l-4,gy+4);
}
for(var i=0;i<n;i++){
var d=data[i];
var x=bx(i);
var isUp=d.c>=d.o;
ctx.strokeStyle=isUp?'#ea4335':'#34a853';
ctx.fillStyle=isUp?'#ea4335':'#34a853';
ctx.lineWidth=1;
ctx.beginPath();ctx.moveTo(x+bw/2,px(d.h));ctx.lineTo(x+bw/2,px(d.l));ctx.stroke();
ctx.fillRect(x,Math.min(px(d.o),px(d.c)),bw,Math.max(1,Math.abs(px(d.o)-px(d.c))));
if(i%8===0){
ctx.fillStyle='#888';ctx.font='9px sans-serif';ctx.textAlign='center';
ctx.fillText(d.d,x+bw/2,H-6);
}
}
ctx.strokeStyle='#1a73e8';ctx.lineWidth=1.5;

for(var i=4;i<n;i++){
var s=0;for(var j=i-4;j<=i;j++)s+=data[j].c;
ma5.push({i:i,v:s/5});
}
ctx.beginPath();
for(var k=0;k<ma5.length;k++){
var x=bx(ma5[k].i)+bw/2;
var y=px(ma5[k].v);
if(k===0)ctx.moveTo(x,y);else ctx.lineTo(x,y);
}
ctx.stroke();
ctx.strokeStyle='#f57c00';ctx.lineWidth=1;
var ma20=[];
for(var i=19;i<n;i++){
var s=0;for(var j=i-19;j<=i;j++)s+=data[j].c;
ma20.push({i:i,v:s/20});
}
ctx.beginPath();
for(var k=0;k<ma20.length;k++){
var x=bx(ma20[k].i)+bw/2;
var y=px(ma20[k].v);
if(k===0)ctx.moveTo(x,y);else ctx.lineTo(x,y);
}
ctx.stroke();
ctx.fillStyle='#1a73e8';ctx.font='10px sans-serif';ctx.textAlign='left';
ctx.fillText('MA5',pad.l+2,pad.t+12);
ctx.fillStyle='#f57c00';
ctx.fillText('MA20',pad.l+32,pad.t+12);
})();
